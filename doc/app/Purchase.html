def to_s
  if cancelled_at
    status = ( completed_at ? 'reversed' : 'cancelled' ) + ' at ' + I18n.l(cancelled_at, format: :long)
  elsif completed_at
    status = 'completed at ' + I18n.l(completed_at, format: :long)
  else
    status = 'pending'
  end

  "#{currency} #{amount} (#{status})"
end
